Процедура ДанныеДляОтображенияПриАктивизацииСтроки(Элемент)
	ТекущиеДанные = Элемент.ТекущиеДанные;
	Если ТекущиеДанные = Неопределено Тогда
		ТабличныйДокумент = Новый ТабличныйДокумент;
		Элементы.ВариантыОтображения.ТекущаяСтраница = Элементы.ВариантОтображения_ТабличныйДокумент;
		Возврат;
	КонецЕсли;
	
	Данные = ТекущиеДанные.Данные;
	Если ТипЗнч(Данные) = Тип("ТабличныйДокумент") Тогда
		ТабличныйДокумент = Данные;
		Элементы.ВариантыОтображения.ТекущаяСтраница = Элементы.ВариантОтображения_ТабличныйДокумент;
	Иначе
		Строка = Данные;
		Элементы.ВариантыОтображения.ТекущаяСтраница = Элементы.ВариантОтображения_Строка;
	КонецЕсли;
